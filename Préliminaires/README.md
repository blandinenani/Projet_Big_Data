# Structure

releases
└── release
    ├── title
    ├── notes
    ├── videos
    ├── labels
    ├── styles
    ├── country
    ├── master_id
    ├── formats
    ├── artists
    │   └── artist
    │       ├── id
    │       ├── role
    │       ├── anv
    │       └── name
    ├── tracklist
    ├── released
    ├── extraartists
    │   └── artist
    │       ├── id
    │       ├── role
    │       ├── anv
    │       └── name
    ├── identifiers
